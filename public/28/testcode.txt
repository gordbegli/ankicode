def test_groupAnagrams():
    # Test case 1
    result = groupAnagrams(["eat","tea","tan","ate","nat","bat"])
    expected = [["bat"],["nat","tan"],["ate","eat","tea"]]
    assert sorted([sorted(group) for group in result]) == sorted([sorted(group) for group in expected]), "Test case 1 failed"

    # Test case 2
    result = groupAnagrams([""])
    expected = [[""]]
    assert result == expected, "Test case 2 failed"

    # Test case 3
    result = groupAnagrams(["a"])
    expected = [["a"]]
    assert result == expected, "Test case 3 failed"

    # Test case 4: Added additional edge case where all strings are identical
    result = groupAnagrams(["abc","abc","abc"])
    expected = [["abc","abc","abc"]]
    assert result == expected, "Test case 4 failed"

    # Test case 5: Added large input to test performance
    large_input = ["a" * i for i in range(1, 101)] * 100
    large_output = [["a" * i for i in range(1, 101)]] * 100
    result = groupAnagrams(large_input)
    assert sorted([sorted(group) for group in result]) == sorted([sorted(group) for group in large_output]), "Test case 5 failed"

    print("All test cases passed!")

# Run the tests
print("Running test cases for groupAnagrams...")
test_groupAnagrams()
